CREATE TABLE "SYSTEM"."MENUITEM" 
   (	"ITEMID" NUMBER(10,0) NOT NULL ENABLE, 
	"ITEMNAME" VARCHAR2(20 BYTE), 
	"CATEGORY" VARCHAR2(20 BYTE), 
	"PRICE" FLOAT(126), 
	"STATUS" VARCHAR2(20 BYTE), 
	"RESTAURANT" VARCHAR2(20 BYTE), 
	 CONSTRAINT "MENUITEM_PK" PRIMARY KEY ("ITEMID"))
	 
	 CREATE SEQUENCE  "SYSTEM"."ITEMID_SEQ"  MINVALUE 1 MAXVALUE 1000 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE ;
