CREATE TABLE "SYSTEM"."PRODUCT" 
   (	"PID" NUMBER(10,0) NOT NULL ENABLE, 
	"PNAME" VARCHAR2(20 BYTE), 
	"PRICE" FLOAT(126), 
	"QTY" FLOAT(126), 
	"STATUS" VARCHAR2(10 BYTE), 
	 CONSTRAINT "PRODUCT_PK" PRIMARY KEY ("PID"))
	 
	  CREATE SEQUENCE  "SYSTEM"."PRODID_SEQ"  MINVALUE 1 MAXVALUE 10000 INCREMENT BY 1 START WITH 100 CACHE 20 NOORDER  NOCYCLE ;
